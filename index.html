How to use Nowtify in my project ?

#1 Format your notifications
let notifications = [
     {
        message     : 'This is the first notification.', // Your message.
        type        : 'success',                        // success, info, warning, danger
        dismissible : true                              // Default: false
    },
     {
        message     : 'This is the second notification',
        type        : 'warning',
    } 
];
#2 Pass your notifications to Nowtify initialized on an DOM element (indicate element's ID)
Nowtify.show( notifications, {
  containerID   : 'ID_CONTAINER',
  displayTimeout: 6000,         // Default 5000ms
  dismissible   :  false        // Default false
});

#3 ... No step 3 cause it's done !
